# The Empty search
# Searches all data from all indexes
GET /_search


# By default, search requests do not timeout
# but you can define if the request should timeout after some time
GET /_search?timeout=10ms


# Search all types in the gb index
GET /gb/_search


# Search all types in the gb and us indices
GET /gb,us/_search


# Search all types in any indices beginning with g or beginning with u
GET /g*,u*/_search

# PAGINATION
GET /_search?size=5
GET /_search?size=5&from=5
GET /_search?size=5&from=10


# SEARCH LITE
GET /_all/_search?q=tweet:elasticsearch


# Search Lite Query (URL encoded):
# +name:john +tweet:mary
GET /_search?q=%2Bname%3Ajohn %2Btweet%3Amary


# The _all field
GET /_search?q=mary


# More complicated query
# +name:(mary john) +date:>2014-09-10 +(aggregations geo)
GET /_search?q=%2Bname%3A(mary+john) %2Bdate%3A%3E2014-09-10 %2B(aggregations geo)
